            <!-- Tombol Tambah Kategori -->

    <script>
        // KONFIGURASI: Masukkan URL Apps Script Anda di sini
        const SCRIPT_URL = 'GANTI_DENGAN_URL_APPS_SCRIPT_ANDA'; 

        let categoryCount = 0;

        // Inisialisasi: Tambah 1 blok kategori saat load
        window.onload = function() {
            addCategoryBlock();
        };

        function addCategoryBlock() {
            categoryCount++;
            const container = document.getElementById('categoriesContainer');
            
            const blockHTML = `
            <div class="category-block bg-gray-50 p-4 rounded border border-gray-200 relative" id="cat-${categoryCount}">
                <div class="flex justify-between items-center mb-4">
                    <h3 class="font-bold text-lg text-blue-800">Kategori #${categoryCount}</h3>
                    ${categoryCount > 1 ? `<button type="button" onclick="removeCategory(${categoryCount})" class="text-red-500 text-sm hover:underline">Hapus</button>` : ''}
                </div>

                <div class="mb-4">
                    <label class="block text-gray-700 text-sm font-bold mb-2">Pilih Kategori</label>
                    <select onchange="toggleFields(this)" class="cat-select w-full border border-gray-300 rounded p-2 bg-white">
                        <option value="">-- Pilih Kategori --</option>
                        <option value="1">1. Pergantian Part</option>
                        <option value="2">2. Maintenance Monthly</option>
                        <option value="3">3. CheckUp Daily</option>
                    </select>
                </div>

                <!-- Dynamic Fields Container -->
                <div class="dynamic-fields space-y-4"></div>
            </div>
            `;
            
            container.insertAdjacentHTML('beforeend', blockHTML);
        }

        function removeCategory(id) {
            document.getElementById(`cat-${id}`).remove();
        }

        function toggleFields(selectElement) {
            const categoryValue = selectElement.value;
            const parentBlock = selectElement.closest('.category-block');
            const fieldsContainer = parentBlock.querySelector('.dynamic-fields');
            
            fieldsContainer.innerHTML = ''; // Reset fields

            if (!categoryValue) return;

            let fieldsHTML = '';

            // Logic Field Berdasarkan Kategori
            if (categoryValue === '1' || categoryValue === '2') {
                // 1. Pergantian Part & 2. Maintenance Monthly
                fieldsHTML = `
                    <div class="grid grid-cols-1 md:grid-cols-2 gap-4">
                        <div class="col-span-1 md:col-span-2">
                            <label class="block text-gray-700 text-sm font-bold mb-1">Task / Deskripsi Pekerjaan</label>
                            <textarea name="task" class="w-full border p-2 rounded" rows="2" required></textarea>
                        </div>
                        
                        <!-- Sebelum -->
                        <div class="bg-white p-2 rounded border">
                            <label class="block text-xs font-bold text-gray-500 mb-1">FOTO SEBELUM</label>
                            <input type="file" name="imgBefore" accept="image/*" class="w-full text-sm text-gray-500 file:mr-4 file:py-2 file:px-4 file:rounded-full file:border-0 file:text-sm file:font-semibold file:bg-blue-50 file:text-blue-700 hover:file:bg-blue-100">
                            <input type="time" name="timeBefore" class="mt-2 w-full border p-1 rounded text-sm">
                        </div>

                        <!-- Sesudah -->
                        <div class="bg-white p-2 rounded border">
                            <label class="block text-xs font-bold text-gray-500 mb-1">FOTO SESUDAH</label>
                            <input type="file" name="imgAfter" accept="image/*" class="w-full text-sm text-gray-500 file:mr-4 file:py-2 file:px-4 file:rounded-full file:border-0 file:text-sm file:font-semibold file:bg-green-50 file:text-green-700 hover:file:bg-green-100">
                            <input type="time" name="timeAfter" class="mt-2 w-full border p-1 rounded text-sm">
                        </div>
                    </div>
                `;
            } else if (categoryValue === '3') {
                // 3. CheckUp Daily
                fieldsHTML = `
                    <div class="grid grid-cols-1 md:grid-cols-2 gap-4">
                        <div class="col-span-1 md:col-span-2">
                            <label class="block text-gray-700 text-sm font-bold mb-1">Task / Item Check</label>
                            <textarea name="task" class="w-full border p-2 rounded" rows="2" required></textarea>
                        </div>
                        
                        <div class="bg-white p-2 rounded border col-span-1 md:col-span-2">
                            <label class="block text-xs font-bold text-gray-500 mb-1">FOTO CHECK & TIMESTAMP</label>
                            <div class="flex gap-2">
                                <input type="file" name="imgCheck" accept="image/*" class="w-full text-sm text-gray-500 file:mr-4 file:py-2 file:px-4 file:rounded-full file:border-0 file:text-sm file:font-semibold file:bg-purple-50 file:text-purple-700 hover:file:bg-purple-100">
                                <input type="time" name="timeCheck" class="border p-2 rounded w-32">
                            </div>
                        </div>
                    </div>
                `;
            }

            fieldsContainer.innerHTML = fieldsHTML;
        }

        // --- FUNGSI ACTION ---

        function collectData() {
            const outlet = document.getElementById('outlet').value;
            if(!outlet) { alert("Pilih Outlet terlebih dahulu!"); return null; }

            const blocks = document.querySelectorAll('.category-block');
            let entries = [];

            blocks.forEach(block => {
                const select = block.querySelector('.cat-select');
                const category = select.value;
                if(!category) return; // Skip jika kategori belum dipilih

                const task = block.querySelector('[name="task"]').value;
                if(!task) { alert("Isi Task pada salah satu kategori!"); throw new Error("Task kosong"); }

                let entry = {
                    category: select.options[select.selectedIndex].text,
                    task: task
                };

                // Ambil data berdasarkan tipe kategori
                if (category === '1' || category === '2') {
                    entry.imgBefore = block.querySelector('[name="imgBefore"]').value ? "Ada Foto" : "-";
                    entry.timeBefore = block.querySelector('[name="timeBefore"]').value || "-";
                    entry.imgAfter = block.querySelector('[name="imgAfter"]').value ? "Ada Foto" : "-";
                    entry.timeAfter = block.querySelector('[name="timeAfter"]').value || "-";
                } else if (category === '3') {
                    entry.imgCheck = block.querySelector('[name="imgCheck"]').value ? "Ada Foto" : "-";
                    entry.timeCheck = block.querySelector('[name="timeCheck"]').value || "-";
                }
                entries.push(entry);
            });

            if(entries.length === 0) { alert("Pilih minimal satu kategori dan isi task!"); return null; }

            return { outlet, entries };
        }

        function sendData() {
            const data = collectData();
            if (!data) return;

            const btnSubmit = document.querySelector('button[onclick="saveDraft()"]');
            const originalText = btnSubmit.innerText;
            btnSubmit.innerText = "Menyimpan...";
            btnSubmit.disabled = true;

            fetch(SCRIPT_URL, {
                method: 'POST',
                body: JSON.stringify(data)
            })
            .then(response => {
                alert("Data berhasil disimpan ke Spreadsheet!");
                // resetForm(); // Opsional: reset setelah simpan
            })
            .catch(error => {
                console.error('Error!', error.message);
                alert("Gagal menyimpan. Cek koneksi atau URL Script.");
            })
            .finally(() => {
                btnSubmit.innerText = originalText;
                btnSubmit.disabled = false;
            });
        }

        function saveDraft() {
            // Dalam konteks web statis tanpa database lokal, "Save Draft" 
            // kita implementasikan dengan langsung mengirim ke Google Sheet 
            // tapi menandainya sebagai draft (bisa dimodifikasi di Apps Script jika perlu)
            // Atau cukup simpan ke LocalStorage browser
            const data = collectData();
            if(data) {
                localStorage.setItem('maintenanceDraft', JSON.stringify(data));
                alert("Draft tersimpan di browser (Local Storage)!");
                // Jika ingin langsung simpan ke sheet juga, panggil sendData()
                sendData(); 
            }
        }

        function resetForm() {
            if(confirm("Apakah Anda yakin ingin mereset formulir?")) {
                document.getElementById('mainForm').reset();
                document.getElementById('categoriesContainer').innerHTML = '';
                categoryCount = 0;
                addCategoryBlock();
                localStorage.removeItem('maintenanceDraft');
            }
        }

        function convertPDF() {
            const element = document.getElementById('printableArea');
            const now = new Date();
            const filename = `Laporan_${now.toISOString().slice(0,19).replace(/:/g,"-")}.pdf`;
            
            const opt = {
                margin:       0.5,
                filename:     filename,
                image:        { type: 'jpeg', quality: 0.98 },
                html2canvas:  { scale: 2 },
                jsPDF:        { unit: 'in', format: 'letter', orientation: 'portrait' }
            };

            // Sembunyikan tombol hapus kategori sebelum print/pdf
            document.querySelectorAll('.no-print').forEach(el => el.style.display = 'none');
            
            html2pdf().set(opt).from(element).save().then(() => {
                 document.querySelectorAll('.no-print').forEach(el => el.style.display = '');
            });
        }
